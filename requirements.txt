def show_menu() -> str:
    return cleandoc(
        """
        1. Zakoduj podany wyraz przy użyciu wybranego kodowania ROT!
        2. Odkoduj wyraz!
        3. Wyjdź z programu!
        """
    )


def main():
    while True:
        print(show_menu())
        choice: int = int(input("> "))
        if choice == 1:
            ROTBase().show_menu()
        elif choice == 4:
            break


if __name__ == '__main__':
    main()
